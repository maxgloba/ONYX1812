					<link href="css/jquery-ui.css" rel="stylesheet" type="text/css"/>
					<script src="js/jquery-ui.min.js"></script>

					$(".barFirst").slider({
						value: 75,
						min: 0,
						max: 100,
						step: 1,
						orientation: "vertical",
						create: function(event, ui){
							val = $(".barFirst").slider("value");
							valRes = 100 - val;
							$(".barFirstVal").html(val);
							$(".barFirstShadow").css("height", valRes+"%" );
						},
						slide:function(event, ui){
							valUiRes = 100 - ui.value;
							$(".barFirstVal").html(ui.value);
							$(".barFirstShadow").css("height", valUiRes+"%" );
						}
					});
