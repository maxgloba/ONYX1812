					$('.btngo').click(function(){
						$(".num").each(function(){
							var number = jQuery(this).attr('number');
							$(this).animate({num:number},{
								duration:1000,
								step:function (num){
									this.innerHTML=(num).toFixed( 0 );
								}
							});
						});
					});
					$('.btnback').click(function(){
						$(".num").each(function(){
							$(this).animate({num:0},{
								duration:200,
								step:function (num){
									this.innerHTML=(num).toFixed( 0 );
								}
							});
						});
					});
